// src/jobs/emailDigestJob.js // ============================================================ // FEATURE 3: WEEKLY EMAIL DIGEST // Runs every Sunday at 6pm. // Sends a personalised weekly summary email to each user. // // Email delivery: Google SMTP first → Resend as fallback. // Install: npm install resend nodemailer // // .env vars needed: // GMAIL_USER=your@gmail.com // GMAIL_APP_PASSWORD=xxxx xxxx xxxx xxxx (Google App Password) // RESEND_API_KEY=re_... // EMAIL_DIGEST_FROM=Clutch AI // FRONTEND_URL=https://yourapp.com // // DB requirements (run before first use): // ALTER TABLE users // ADD COLUMN IF NOT EXISTS email_digest_enabled boolean DEFAULT true, // ADD COLUMN IF NOT EXISTS last_digest_sent_at timestamptz, // ADD COLUMN IF NOT EXISTS digest_email text; // ============================================================ import supabaseAdmin from '../config/supabase.js'; import { callWithFallback } from '../services/multiProvider.js'; import { recordTokenUsage } from '../services/tokenTracker.js'; import { EMAIL_DIGEST_FROM, BATCH_DELAY_MS } from '../config/constants.js'; import nodemailer from 'nodemailer'; import { Resend } from 'resend'; const sleep = (ms) => new Promise(r => setTimeout(r, ms)); const BATCH_SIZE = 10; const FRONTEND_URL = process.env.FRONTEND_URL || 'https://app.clutch.ai'; const logJob = async (name, status, data = {}) => { await supabaseAdmin.from('job_logs').insert({ job_name: name, status, ...data }).catch(() => {}); }; // ────────────────────────────────────────── // BUILD SMTP / RESEND CLIENTS LAZILY // ────────────────────────────────────────── let gmailTransport = null; let resendClient = null; const getGmailTransport = () => { if (!gmailTransport && process.env.GMAIL_USER && process.env.GMAIL_APP_PASSWORD) { gmailTransport = nodemailer.createTransport({ host: 'smtp.gmail.com', port: 587, secure: false, auth: { user: process.env.GMAIL_USER, pass: process.env.GMAIL_APP_PASSWORD, }, }); } return gmailTransport; }; const getResendClient = () => { if (!resendClient && process.env.RESEND_API_KEY) { resendClient = new Resend(process.env.RESEND_API_KEY); } return resendClient; }; // ────────────────────────────────────────── // MAIN JOB ENTRY POINT // ────────────────────────────────────────── export const runEmailDigestJob = async () => { const startTime = Date.now(); console.log(`[EmailDigest] Starting ${new Date().toISOString()}`); await logJob('email_digest', 'started'); let sent = 0, failed = 0; try { const { data: users } = await supabaseAdmin .from('users') .select('id, name, email, digest_email, archetype, product_description, target_audience, email_digest_enabled') .eq('onboarding_completed', true) .eq('is_deleted', false) .eq('email_digest_enabled', true) .not('email', 'is', null); if (!users?.length) { await logJob('email_digest', 'completed', { sent: 0, failed: 0, duration_ms: Date.now() - startTime }); return; } console.log(`[EmailDigest] Sending to ${users.length} users`); // Process in batches for (let i = 0; i < users.length; i += BATCH_SIZE) { const batch = users.slice(i, i + BATCH_SIZE); const results = await Promise.allSettled(batch.map(user => sendDigestForUser(user))); results.forEach((r, idx) => { if (r.status === 'fulfilled') sent++; else { failed++; console.error(`[EmailDigest] Failed for ${batch[idx].email}:`, r.reason?.message); } }); if (i + BATCH_SIZE < users.length) await sleep(BATCH_DELAY_MS); } await logJob('email_digest', 'completed', { sent, failed, duration_ms: Date.now() - startTime }); console.log(`[EmailDigest] Done — ${sent} sent, ${failed} failed`); } catch (err) { console.error('[EmailDigest] Fatal:', err.message); await logJob('email_digest', 'failed', { error_message: err.message, duration_ms: Date.now() - startTime }); } }; // ────────────────────────────────────────── // SEND DIGEST FOR ONE USER // ────────────────────────────────────────── const sendDigestForUser = async (user) => { const toEmail = user.digest_email || user.email; const sevenDaysAgo = new Date(Date.now() - 7 * 86400000).toISOString().split('T')[0]; const today = new Date().toISOString().split('T')[0]; // Gather weekly stats in parallel const [metricsResult, goalsResult, pipelineResult, perfResult] = await Promise.allSettled([ supabaseAdmin .from('daily_metrics') .select('messages_sent, positive_outcomes, opportunities_shown') .eq('user_id', user.id) .gte('date', sevenDaysAgo) .lte('date', today), supabaseAdmin .from('user_goals') .select('goal_text, current_value, target_value, target_unit, status') .eq('user_id', user.id) .eq('status', 'active') .limit(2), supabaseAdmin .from('opportunities') .select('stage, deal_value') .eq('user_id', user.id) .not('stage', 'in', '("new","closed_lost")'), supabaseAdmin .from('user_performance_profiles') .select('positive_rate, total_sent') .eq('user_id', user.id) .single(), ]); const metrics = metricsResult.status === 'fulfilled' ? metricsResult.value.data || [] : []; const goals = goalsResult.status === 'fulfilled' ? goalsResult.value.data || [] : []; const pipeline = pipelineResult.status === 'fulfilled' ? pipelineResult.value.data || [] : []; const perf = perfResult.status === 'fulfilled' ? perfResult.value.data : null; // Aggregate stats const sentCount = metrics.reduce((s, d) => s + (d.messages_sent || 0), 0); const positiveCount = metrics.reduce((s, d) => s + (d.positive_outcomes || 0), 0); const shownCount = metrics.reduce((s, d) => s + (d.opportunities_shown || 0), 0); const replyRatePct = sentCount > 0 ? Math.round((positiveCount / sentCount) * 100) : 0; // Compute streak: count consecutive days (going backwards from today) that had at least 1 message sent const metricsByDate = {}; metrics.forEach(d => { metricsByDate[d.date] = d; }); let streakDays = 0; for (let i = 0; i < 30; i++) { const d = new Date(Date.now() - i * 86400000).toISOString().split('T')[0]; if (metricsByDate[d]?.messages_sent > 0) { streakDays++; } else if (i > 0) { // Allow today to be 0 (day not finished yet) only if yesterday had activity break; } } const pipelineValue = pipeline.reduce((s, o) => s + (o.deal_value || 0), 0); const stageCounts = pipeline.reduce((acc, o) => { acc[o.stage] = (acc[o.stage] || 0) + 1; return acc; }, {}); const isQuietWeek = sentCount === 0 && shownCount === 0; // Generate Groq commentary — Groq only (no Perplexity for digest per requirements) let commentary = ''; try { const goalSummaries = goals.map(g => `${g.goal_text}: ${g.current_value || 0}/${g.target_value || '?'} ${g.target_unit || ''}` ).join(', ') || 'none set'; const prompt = isQuietWeek ? `Write 2-3 sentences of warm, motivating coaching for a founder who had a quiet week (sent 0 messages, found 0 leads). Encourage them to start small next week. Product: ${user.product_description || 'not specified'}. Sound like a real coach, not a newsletter.` : `You are an AI sales coach writing a brief weekly summary for a founder. Write 2-3 sentences of coaching commentary based on their week's performance. Be specific to their numbers. Be encouraging but honest. Mention one concrete thing they should focus on next week. Sound like a real coach, not a newsletter. Never use words like "leverage", "synergy", or "unlock". User profile: ${user.product_description || 'not specified'} | archetype: ${user.archetype || 'seller'} This week: ${sentCount} messages sent, ${positiveCount} positive replies, reply rate: ${replyRatePct}%, pipeline value: $${pipelineValue} Goals: ${goalSummaries} Respond with ONLY the commentary text, 2-3 sentences.`; const { content, tokens_in, tokens_out } = await callWithFallback({ systemPrompt: 'You write concise, personalised coaching commentary. No markdown, no formatting.', messages: [{ role: 'user', content: prompt }], temperature: 0.7, maxTokens: 200, }); await recordTokenUsage(user.id, 'groq', tokens_in, tokens_out); commentary = content.trim(); } catch (err) { console.warn(`[EmailDigest] Groq failed for ${user.id}, using generic comment`); commentary = isQuietWeek ? 'Every week is a fresh start. Even one outreach message puts you ahead of doing nothing. You have everything you need to make next week count.' : `You sent ${sentCount} messages this week. Keep logging your outcomes so Clutch can learn what works best for you.`; } // Build HTML email const html = buildEmailHtml({ user, sentCount, positiveCount, replyRatePct, streakDays, pipelineValue, stageCounts, goals, commentary, isQuietWeek }); const subject = isQuietWeek ? `Let's make next week count 🚀` : `Your week: ${sentCount} messages, ${positiveCount} replies 📈`; // Send: Gmail first → Resend fallback await sendEmail({ to: toEmail, subject, html }); // Update last_digest_sent_at await supabaseAdmin .from('users') .update({ last_digest_sent_at: new Date().toISOString() }) .eq('id', user.id); }; // ────────────────────────────────────────── // EMAIL DELIVERY — Gmail → Resend fallback // ────────────────────────────────────────── const sendEmail = async ({ to, subject, html }) => { const gmail = getGmailTransport(); if (gmail) { try { await gmail.sendMail({ from: EMAIL_DIGEST_FROM, to, subject, html }); return; // Success — done } catch (err) { // Distinguish transient (5xx, timeout) from permanent (4xx auth) errors const isPermanent = err.responseCode >= 400 && err.responseCode < 500; if (isPermanent) { console.error(`[EmailDigest] Gmail permanent error (${err.responseCode}), will NOT fallback:`, err.message); throw err; } console.warn(`[EmailDigest] Gmail transient failure, falling back to Resend:`, err.message); } } // Fallback: Resend const resend = getResendClient(); if (!resend) throw new Error('No email provider configured (GMAIL_APP_PASSWORD and RESEND_API_KEY both missing)'); const { error } = await resend.emails.send({ from: EMAIL_DIGEST_FROM, to, subject, html }); if (error) throw new Error(`Resend error: ${error.message}`); }; // ────────────────────────────────────────── // HTML EMAIL BUILDER — table-based layout for email client compatibility // ────────────────────────────────────────── const buildEmailHtml = ({ user, sentCount, positiveCount, replyRatePct, streakDays, pipelineValue, stageCounts, goals, commentary, isQuietWeek }) => { const firstName = user.name?.split(' ')[0] || 'there'; const settingsUrl = `${FRONTEND_URL}/settings?tab=notifications`; const ctaUrl = FRONTEND_URL; const goalRows = goals.length ? goals.map(g => { const pct = g.target_value ? Math.min(100, Math.round(((g.current_value || 0) / g.target_value) * 100)) : null; const barWidth = pct !== null ? pct : 0; return `

${g.goal_text}

${pct !== null ? `
 
${pct}%
` : '

No target set

'} `; }).join('') : `No active goals this week.`; const stageItems = Object.entries(stageCounts) .map(([stage, count]) => `${count} ${stage.replace('_', '/')}`) .join(' · ') || 'No active pipeline'; return `

Clutch AI

Your week in review

Hey ${firstName} — here's how your week looked.

${sentCount}

Sent

${positiveCount}

Replies

${replyRatePct}%

Reply Rate

${streakDays}🔥

Day Streak

Pipeline

${stageItems}

Active Goals

${goalRows}

🧠 Your Coach Says

${commentary}

Open Clutch →

You're receiving this because weekly digests are enabled. Unsubscribe

`; };
