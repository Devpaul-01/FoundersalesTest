import { asyncHandler } from '../middleware/errorHandler.js'; import supabaseAdmin from '../config/supabase.js'; import groqService from '../services/groq.js'; import { notifyUser } from '../services/notifications.js'; import { callWithFallback } from '../services/multiProvider.js'; import { recordTokenUsage } from '../services/tokenTracker.js'; const router = Router(); // ───────────────────────────────────────────────────────────────────────────── // GET /api/growth/feed // Unified feed: growth_cards + recent opportunities, merged and sorted. // Growth cards are primary for non-Seller archetypes. // ───────────────────────────────────────────────────────────────────────────── router.get('/feed', asyncHandler(async (req, res) => { const userId = req.user.id; } // Build unified feed items with type tags const feedItems = []; // For seller archetype: interleave cards + opps // For others: cards first, then opps at the bottom const SELLER_ARCHETYPES = ['seller', 'freelancer', 'builder']; const cardItems = (cards || []).map(c => ({ ...c, feed_type: 'growth_card' })); let oi = 0, ci = 0; while (oi < oppItems.length || ci < cardItems.length) { if (oi < oppItems.length) feedItems.push(oppItems[oi++]); if (oi < oppItems.length) feedItems.push(oppItems[oi++]); if (ci < cardItems.length) feedItems.push(cardItems[ci++]); } } else { // Non-seller: growth cards first, then any opportunities (cards || []).forEach(c => feedItems.push({ ...c, feed_type: 'growth_card' })); (opps || []).forEach(o => feedItems.push({ ...o, feed_type: 'opportunity' })); } res.json({ feed: feedItems.slice(0, limit), goals: goals || [], archetype, has_opportunities: (opps?.length || 0) > 0, has_cards: (cards?.length || 0) > 0, total_active_opps: opps?.length || 0, }); })); // ───────────────────────────────────────────────────────────────────────────── // POST /api/growth/cards/:id/read // ───────────────────────────────────────────────────────────────────────────── router.post('/cards/:id/read', asyncHandler(async (req, res) => { await supabaseAdmin .from('growth_cards') .update({ is_read: true }) .eq('id', req.params.id) .eq('user_id', req.user.id); res.json({ success: true }); })); // ───────────────────────────────────────────────────────────────────────────── // POST /api/growth/cards/:id/dismiss // ───────────────────────────────────────────────────────────────────────────── router.post('/cards/:id/dismiss', asyncHandler(async (req, res) => { await supabaseAdmin .from('growth_cards') .update({ is_dismissed: true }) .eq('id', req.params.id) .eq('user_id', req.user.id); res.json({ success: true }); })); // ───────────────────────────────────────────────────────────────────────────── // GET /api/growth/checkin/today // Returns today's check-in. } // Get the check-in record const { data: checkIn } = await supabaseAdmin .from('daily_check_ins') .select('*') .eq('user_id', userId) .eq('date', today) .single(); if (!checkIn) { return res.status(404).json({ error: 'NOT_FOUND', message: 'No check-in found for today' }); } // Get active goals for response context const { data: goals } = await supabaseAdmin .from('user_goals') .select('goal_text, current_value, target_value, target_unit') .eq('user_id', userId) .eq('status', 'active') .limit(3); // Generate AI response to the check-in const { response_text, next_tip_seed } = await groqService.generateCheckInResponse( req.user, req.user.archetype || 'seller', checkIn.questions, answers, goals || [] ); // Save answers + AI response await supabaseAdmin .from('daily_check_ins') .update({ answers, mood_score: mood_score || null, ai_response: response_text, processed_at: new Date().toISOString() }) .eq('id', checkIn.id); // Update last check-in timestamp on user await supabaseAdmin .from('users') .update({ last_check_in_at: new Date().toISOString() }) .eq('id', userId); // Generate tomorrow's personalized tip card based on the check-in answers // (async, don't block the response) generateTipFromCheckIn(userId, req.user, answers, next_tip_seed, goals).catch(err => console.error('[CheckIn] Tip generation failed:', err.message) ); res.json({ success: true, ai_response: response_text, message: 'Check-in saved. Your growth tip for tomorrow has been updated.' }); })); // ───────────────────────────────────────────────────────────────────────────── // GOALS // ───────────────────────────────────────────────────────────────────────────── router.get('/goals', asyncHandler(async (req, res) => { const { data: goals } = await supabaseAdmin .from('user_goals') .select('*') .eq('user_id', req.user.id) .order('created_at', { ascending: false }); res.json({ goals: goals || [] }); })); router.post('/goals', asyncHandler(async (req, res) => { const { goal_text, goal_type, target_value, target_unit, target_date } = req.body; if (!goal_text?.trim()) { return res.status(400).json({ error: 'VALIDATION_ERROR', message: 'goal_text required' }); } const { data: goal } = await supabaseAdmin .from('user_goals') .insert({ user_id: req.user.id, goal_text: goal_text.trim(), goal_type: goal_type || 'custom', target_value: target_value || null, target_unit: target_unit || null, target_date: target_date || null, }) .select() .single(); res.json({ success: true, goal }); })); router.put('/goals/:id', asyncHandler(async (req, res) => { const { current_value, status, goal_text } = req.body; const updates = { updated_at: new Date().toISOString() }; if (current_value !== undefined) updates.current_value = current_value; if (status !== undefined) updates.status = status; if (goal_text !== undefined) updates.goal_text = goal_text; const { data: goal } = await supabaseAdmin .from('user_goals') .update(updates) .eq('id', req.params.id) .eq('user_id', req.user.id) .select() .single(); res.json({ success: true, goal }); })); router.delete('/goals/:id', asyncHandler(async (req, res) => { await supabaseAdmin .from('user_goals') .delete() .eq('id', req.params.id) .eq('user_id', req.user.id); res.json({ success: true }); })); // ───────────────────────────────────────────────────────────────────────────── // GOAL NOTES — Feature 6 // POST /api/growth/goals/:goalId/notes — log a progress note, get AI coaching // GET /api/growth/goals/:goalId/notes — fetch note history // DELETE /api/growth/goals/:goalId/notes/:noteId — delete a note // ───────────────────────────────────────────────────────────────────────────── router.post('/goals/:goalId/notes', asyncHandler(async (req, res) => { const { goalId } = req.params; const { note_text } = req.body; if (!note_text?.trim()) { return res.status(400).json({ error: 'VALIDATION_ERROR', message: 'Note text is required' }); } // Fetch goal and verify ownership const { data: goal } = await supabaseAdmin .from('user_goals') .select('*') .eq('id', goalId) .eq('user_id', req.user.id) .single(); if (!goal) { return res.status(404).json({ error: 'NOT_FOUND', message: 'Goal not found' }); } // Build Groq prompt const systemPrompt = ` You are an AI coach analyzing a founder's goal progress note. Goal: "${goal.goal_text}" Target: ${goal.target_value ?? 'not set'} ${goal.target_unit ?? ''} Current progress: ${goal.current_value ?? 0} / ${goal.target_value ?? '?'} ${goal.target_unit ?? ''} User's business: ${req.user.product_description ?? 'not provided'} The user just logged this note: "${note_text.trim()}" You must respond ONLY as JSON with this exact shape: { "progress_delta": , "coaching_response": "<2-4 sentences: 1) acknowledge what they did specifically, 2) relate it to their goal progress with the actual numbers, 3) give ONE specific concrete next action. Keep logging your progress — patterns will emerge over time.', needs_tip_card: false, tip_context: null, }; } const delta = parseFloat(parsed.progress_delta) || 0; // Atomic goal progress update — cap at 0 to prevent negative values if (delta !== 0) { await supabaseAdmin.rpc('increment_goal_progress', { p_goal_id: goalId, p_delta: delta, }); } // Compute new value and check for completion const newValue = Math.max(0, (goal.current_value ?? 0) + delta); if (goal.target_value && newValue >= goal.target_value && goal.status === 'active') { await supabaseAdmin .from('user_goals') .update({ status: 'completed', completed_at: new Date().toISOString() }) .eq('id', goalId); } // Save the note with AI response const { data: note } = await supabaseAdmin .from('goal_notes') .insert({ goal_id: goalId, user_id: req.user.id, note_text: note_text.trim(), ai_response: parsed.coaching_response, progress_delta: delta, }) .select() .single(); // Fire-and-forget tip card generation — do NOT await, user must not wait for this if (parsed.needs_tip_card && parsed.tip_context) { (async () => { try { const tipPrompt = ` Generate a concise growth tip for a founder. Context: ${parsed.tip_context} Their goal: ${goal.goal_text} Their product: ${req.user.product_description ?? ''} Respond ONLY as JSON: { "title": "<10 words max>", "body": "<2-3 sentences of actionable advice>" } `.trim(); const { content: tipContent, tokens_in: tIn, tokens_out: tOut } = await callWithFallback({ systemPrompt: tipPrompt, messages: [{ role: 'user', content: 'Generate the tip.' }], temperature: 0.5, maxTokens: 150, }); await recordTokenUsage(req.user.id, 'groq', tIn, tOut); const tip = JSON.parse(tipContent.replace(/```json|```/g, '').trim()); await supabaseAdmin.from('growth_cards').insert({ user_id: req.user.id, card_type: 'tip', title: tip.title, body: tip.body, action_label: 'Log more progress', action_type: 'internal_goals', priority: 7, expires_at: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), generated_by: 'goal_note_ai', metadata: { goal_id: goalId }, }); } catch (err) { console.error('[GoalNotes] Tip card generation failed (non-fatal):', err.message); } })(); } res.status(201).json({ success: true, note, coaching_response: parsed.coaching_response, progress_delta: delta, new_value: newValue, goal_completed: goal.target_value ? newValue >= goal.target_value : false, }); })); router.get('/goals/:goalId/notes', asyncHandler(async (req, res) => { const { goalId } = req.params; // Verify goal ownership first const { data: goal } = await supabaseAdmin .from('user_goals') .select('id') .eq('id', goalId) .eq('user_id', req.user.id) .single(); if (!goal) { return res.status(404).json({ error: 'NOT_FOUND', message: 'Goal not found' }); } const { data: notes } = await supabaseAdmin .from('goal_notes') .select('id, note_text, ai_response, progress_delta, created_at') .eq('goal_id', goalId) .eq('user_id', req.user.id) .order('created_at', { ascending: false }) .limit(50); res.json({ notes: notes || [] }); })); router.delete('/goals/:goalId/notes/:noteId', asyncHandler(async (req, res) => { await supabaseAdmin .from('goal_notes') .delete() .eq('id', req.params.noteId) .eq('user_id', req.user.id); res.json({ success: true }); })); // ───────────────────────────────────────────────────────────────────────────── // GET /api/growth/history // Returns paginated past daily tips and weekly plans for review. // ───────────────────────────────────────────────────────────────────────────── router.get('/history', asyncHandler(async (req, res) => { const userId = req.user.id; const limit = parseInt(req.query.limit) || 30; const offset = parseInt(req.query.offset) || 0; const type = req.query.type; // 'tips' | 'plans' | undefined (all) let query = supabaseAdmin .from('growth_cards') .select('id, card_type, title, body, action_label, generated_by, created_at, is_read, metadata') .eq('user_id', userId) .order('created_at', { ascending: false }) .range(offset, offset + limit - 1); if (type === 'tips') { query = query.in('card_type', ['tip', 'challenge', 'reflection', 'resource']); } else if (type === 'plans') { query = query.eq('card_type', 'strategy').eq('generated_by', 'ai_weekly'); } const result = await groqService.detectUserArchetype(user, user.onboarding_answers || {}); await supabaseAdmin .from('users') .update({ archetype: result.archetype, archetype_detected_at: new Date().toISOString() }) .eq('id', userId); res.json({ plan: card, cached: false }); })); // ───────────────────────────────────────────────────────────────────────────── // INTERNAL HELPERS // ───────────────────────────────────────────────────────────────────────────── const generateFirstTimeCards = async (userId, user) => { console.log("Generating first time growth cards"); export default router;
